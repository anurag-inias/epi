# The Sudoku checker problem

## Description

Check whether a $9 \times 9$ 2D array representing a partially completed Sudoku is valid. 

## Details

a sudoku matrix is valid if:

- there are duplicates in any of the rows or column.
- each $3 \times 3$ block of the matrix must contain the digits $1$ through $9$ with no duplicates.

    fun isValidSudoku(partialAssignment: MutableList<MutableList<Int>>): Boolean {
      // no duplicates in any row.
      for (row in partialAssignment) {
        val row = row.filter { it > 0 }
        if (row.size != row.distinct().size) return false
      }

      // no duplicates in any column.
      for (col in partialAssignment.indices) {
        val seen = mutableSetOf<Int>()
        for (row in partialAssignment.indices) {
          val cell = partialAssignment[row][col]
          if (cell > 0 && !seen.add(cell)) return false
        }
      }

      // Check each of the 9 blocks.
      for (row in 0..6 step 3) {
        for (col in 0..6 step 3) {
          if (!isValidBlock(partialAssignment, (row to col), (row + 3 to col + 3))) return false
        }
      }

      return true
    }

    fun isValidBlock(
      partialAssignment: MutableList<MutableList<Int>>,
      from: Pair<Int, Int>,
      to: Pair<Int, Int>
    ): Boolean {
      val (fromRow, fromCol) = from
      val (toRow, toCol) = to

      val seen = mutableSetOf<Int>()
      for (row in fromRow until toRow) {
        for (col in fromCol until toCol) {
          val cell = partialAssignment[row][col]
          if (cell > 0 && !seen.add(cell)) return false
        }
      }

      return true
    }
