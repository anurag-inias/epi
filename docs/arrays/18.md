# Rotate a 2D array

## Description

Given a 2D array, rotate it by $90^{\circ}$ clockwise.


## Example

```
1  2  3 
4  5  6 
7  8  9 

becomes

7  4  1 
8  5  2 
9  6  3 
```

    ```kotlin
    fun rotateMatrix(m: MutableList<MutableList<Int>>) {
      val n = m.size
      for (l in 0 until n / 2) {
        val top = l
        val bottom = n - 1 - l
        val left = l
        val right = n - 1 - l
        val side = right - left + 1
        val backup = Array(side) { 0 }

        // Overwrite backed up "top row" with incoming "left column".
        for (i in side - 1 downTo 0) {
          val temp = m[top][left + i]
          m[top][left + i] = m[bottom - i][left]
          backup[i] = temp
        }

        // "left column" with "bottom row".
        for (i in side - 1 downTo 0) {
          m[bottom - i][left] = m[bottom][right - i]
        }

        // "bottom row" with "right column".
        for (i in side - 1 downTo 0) {
          m[bottom][right - i] = m[top + i][right]
        }

        // "right column" with "top row" backed up in the beginning.
        for (i in side - 1 downTo 0) {
          m[top + i][right] = backup[i]
        }
      }
    }
    ```
